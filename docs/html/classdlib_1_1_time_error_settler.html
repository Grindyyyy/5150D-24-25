void opcontrol(){
dlib::Pid<Meters> pid({1,2,3});
// Construct a last_period period
au::Quantity<au::Seconds, double> last_period;
// Construct an TimeErrorSettler controller
dlib::TimeErrorSettler<Meters> pid_settler {
meters(0.01), // error threshold, the maximum error the pid can settle at
milli(seconds)(.2) // derivative threshold, the maximum instantaneous error over time the pid can settle at
};
// Instantiate error and derivative
au::Quantity<Meters, double> error = pid.get_error();
au::Quantity<au::Seconds, double> period = milli(seconds)(pros::millis()) - last_period;
bool settled = pid_settler.is_settled(error, period);
last_period = milli(seconds)(pros::millis());
}
